export const apiClient = async ( url: string, options: RequestInit = {} ): Promise => { const res = await fetch(import.meta.env.VITE_API_URL + url, { credentials: "include", headers: { "Content-Type": "application/json", ...options.headers, }, ...options, }); if (!res.ok) { const error = await res.json(); throw new Error(error.message || "API Error"); } return res.json(); };
